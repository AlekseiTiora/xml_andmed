function searchByName($query){
    global $kaubad;
    $result=array();
    foreach ($kaubad->kaup as $kaup){
        if(substr(strtolower($kaup->nimetus),0,strlen($query))==
            strtolower($query)){
            array_push($result,$kaup);
        }
    }
    return $result;
}
